<script>
    function calculateCost() {
        const fragranceSelect = document.getElementById("fragrance");
        const bottleSelect = document.getElementById("bottle");
        
        const price50ml = parseFloat(fragranceSelect.value);
        const bottleCost = parseFloat(bottleSelect.value);
        const volume = parseFloat(bottleSelect.options[bottleSelect.selectedIndex].getAttribute("data-vol"));
        
        // حساب
        const perfumeCost = (price50ml / 50) * volume;
        const totalBeforeCourier = perfumeCost + bottleCost;
        const courierCost = totalBeforeCourier * 0.04604;
        const grandTotal = Math.round(totalBeforeCourier + courierCost);

        // نتائج دکھائیں
        document.getElementById("perfumeCost").innerText = perfumeCost.toFixed(2) + " روپے";
        document.getElementById("bottleCost").innerText = bottleCost.toFixed(2) + " روپے";
        document.getElementById("courierCost").innerText = courierCost.toFixed(2) + " روپے";
        document.getElementById("grandTotal").innerText = grandTotal + " روپے";
    }

    // صفحہ لوڈ ہونے پر پہلی بار حساب کریں
    window.onload = calculateCost;
</script>
